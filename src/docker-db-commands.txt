# Docker PostgreSQL Check Commands
# Run these in Command Prompt or PowerShell

# 1. Check if container is running
docker ps

# 2. Check schemas in the container
docker exec temp-postgres psql -U postgres -d eshopdb -c "SELECT schema_name FROM information_schema.schemata WHERE schema_name IN ('catalog', 'basket', 'ordering', 'shared', 'messaging', 'identity') ORDER BY schema_name;"

# 3. Check tables in catalog schema
docker exec temp-postgres psql -U postgres -d eshopdb -c "\dt catalog.*"

# 4. Check tables in basket schema  
docker exec temp-postgres psql -U postgres -d eshopdb -c "\dt basket.*"

# 5. Check migration history
docker exec temp-postgres psql -U postgres -d eshopdb -c "SELECT schemaname, tablename FROM pg_tables WHERE tablename = '__EFMigrationsHistory';"

# 6. Get full database structure
docker exec temp-postgres psql -U postgres -d eshopdb -c "\dn+"
